	`market_cap_strict` BIGINT(20) UNSIGNED NULL DEFAULT NULL,
	`price` DECIMAL(30,10) UNSIGNED NULL DEFAULT NULL,
	`circulating_supply` BIGINT(20) UNSIGNED NULL DEFAULT NULL,
	`total_supply` BIGINT(20) UNSIGNED NULL DEFAULT NULL,
	`max_supply` BIGINT(20) UNSIGNED NULL DEFAULT NULL,
	`volume_24h` INT(11) NULL DEFAULT NULL,
	`volume_7d` INT(11) NULL DEFAULT NULL,
	`volume_30d` INT(11) NULL DEFAULT NULL,
	`Change_1h` DECIMAL(20,6) NULL DEFAULT NULL,
	`change_24h` DECIMAL(20,6) NULL DEFAULT NULL,
